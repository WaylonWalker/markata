    ```python
    def make_layout() -> Layout:
        """Define the layout."""
        layout = Layout(name="root")

        layout.split(
            Layout(name="header", size=3),
            Layout(name="main"),
        )
        layout["main"].split_row(
            Layout(name="side", ratio=50),
            Layout(name="mid", ratio=30),
            Layout(name="describe", ratio=20),
        )
        layout["mid"].split(
            Layout(name="server"),
            Layout(name="runner"),
        )
        layout["side"].split(
            Layout(name="plugins"),
        )
        return layout
    ```
