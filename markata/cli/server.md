    ```python
    def find_port(port: int = 8000) -> int:
        """Find a port not in ues starting at given port"""
        import socket

        with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as s:
            if s.connect_ex(("localhost", port)) == 0:
                return find_port(port=port + 1)
            return port
    ```
